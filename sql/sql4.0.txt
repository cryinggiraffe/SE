ALTER TABLE ImportForm ADD date DATE NULL

ALTER TABLE ImportReturnForm ADD date DATE NULL

CREATE TABLE SaleCommodity
(
  id            INT AUTO_INCREMENT
    PRIMARY KEY,
  goodid        VARCHAR(30) NULL,
  name          VARCHAR(30) NULL,
  version       VARCHAR(30) NULL,
  quantity      INT         NULL,
  price         DOUBLE      NULL,
  subtotalprice DOUBLE      NULL,
  remark        VARCHAR(30) NULL,
  saleformid  VARCHAR(30) NULL
);

CREATE TABLE SaleForm
(
  id              VARCHAR(30) NOT NULL
    PRIMARY KEY,
  client          VARCHAR(30) NULL,
  operator        VARCHAR(30) NULL,
  salesman        VARCHAR(30) NULL,
  houseware       VARCHAR(30) NULL,
  tpbfDiscounting DOUBLE      NULL,
  tpafDiscounting DOUBLE      NULL,
  discount        DOUBLE      NULL,
  voucher         DOUBLE      NULL,
  remark          VARCHAR(30) NULL,
  state           VARCHAR(30) NULL,
  date            DATE        NULL
);


CREATE TABLE SaleReturnCommodity
(
  id            INT AUTO_INCREMENT
    PRIMARY KEY,
  goodid        VARCHAR(30) NULL,
  name          VARCHAR(30) NULL,
  version       VARCHAR(30) NULL,
  quantity      INT         NULL,
  price         DOUBLE      NULL,
  subtotalprice DOUBLE      NULL,
  remark        VARCHAR(30) NULL,
  importformid  VARCHAR(30) NULL
);

  id              VARCHAR(30) NOT NULL
