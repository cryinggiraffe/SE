CREATE TABLE Gift
(
  id         INT AUTO_INCREMENT
    PRIMARY KEY,
  giftlistid VARCHAR(30) NULL,
  giftid     VARCHAR(30) NULL,
  name       VARCHAR(30) NULL,
  price      DOUBLE      NULL,
  num        INT         NULL,
  state      VARCHAR(30) NULL,
  date       DATE        NULL
)

CREATE TABLE Cash
(
  id       INT AUTO_INCREMENT
    PRIMARY KEY,
  listid   VARCHAR(30) NULL,
  operator VARCHAR(30) NULL,
  account  VARCHAR(30) NULL,
  name     VARCHAR(30) NULL,
  amount   DOUBLE      NULL,
  remark   VARCHAR(30) NULL,
  sum      DOUBLE      NULL,
  date     DATE        NULL
)
