CREATE TABLE Payment
(
  id       VARCHAR(30) NOT NULL
    PRIMARY KEY,
  client   VARCHAR(20) NULL,
  username VARCHAR(20) NULL,
  account  VARCHAR(20) NULL,
  amount   DOUBLE      NULL,
  remark   VARCHAR(30) NULL,
  sum      DOUBLE      NULL,
  state    VARCHAR(20) NULL,
  date     DATE        NULL
);


CREATE TABLE Receipt
(
  id       VARCHAR(30) NOT NULL
    PRIMARY KEY,
  username VARCHAR(20) NULL,
  account  VARCHAR(20) NULL,
  amount   DOUBLE      NULL,
  remark   VARCHAR(30) NULL,
  sum      DOUBLE      NULL,
  state    VARCHAR(20) NULL
);
