CREATE TABLE Message
(
  id      INT AUTO_INCREMENT
    PRIMARY KEY,
  message VARCHAR(20) NULL
);



CREATE TABLE ImportCommodity
(
  id            INT AUTO_INCREMENT
    PRIMARY KEY,
  goodid        VARCHAR(30) NULL,
  name          VARCHAR(30) NULL,
  version       VARCHAR(30) NULL,
  quantity      INT         NULL,
  price         DOUBLE      NULL,
  subtotalprice DOUBLE      NULL,
  remark        VARCHAR(30) NULL,
  importformid  VARCHAR(30) NULL
);
