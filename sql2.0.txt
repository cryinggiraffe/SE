CREATE TABLE Message
(
  id      INT AUTO_INCREMENT
    PRIMARY KEY,
  message VARCHAR(20) NULL
);


CREATE TABLE Receipt
(
  id       VARCHAR(30) NOT NULL
    PRIMARY KEY,
  username VARCHAR(20) NULL,
  account  VARCHAR(20) NULL,
  amount   DOUBLE      NULL,
  remark   VARCHAR(30) NULL,
  sum      DOUBLE      NULL,
  state    VARCHAR(20) NULL
);

CREATE TABLE Payment
(
  id       VARCHAR(30) NOT NULL
    PRIMARY KEY,
  username VARCHAR(20) NULL,
  account  VARCHAR(20) NULL,
  amount   DOUBLE      NULL,
  remark   VARCHAR(30) NULL,
  sum      DOUBLE      NULL,
  state    VARCHAR(20) NULL
);


CREATE TABLE ImortCommodity
(
  goodid        VARCHAR(30) NOT NULL
    PRIMARY KEY,
  name          VARCHAR(30) NULL,
  version       VARCHAR(20) NULL,
  quantity      INT         NULL,
  price         DOUBLE      NULL,
  subtotalprice DOUBLE      NULL,
  remark        VARCHAR(30) NULL
);
